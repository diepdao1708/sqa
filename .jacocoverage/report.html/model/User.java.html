package model;

    private String phone_number;
    private String email;

    public User() {

    }

        this.user_id = user_id;
        this.account_code = account_code;
        this.password = password;
        this.role = role;
        this.name = name;
        this.cccd = cccd;

        this.email = email;

        this.account_code = account_code;

    public void setPassword(String password) {
        this.password = password;
    }

        return role;

        this.role = role;

        this.name = name;

        this.email = email;
    }
